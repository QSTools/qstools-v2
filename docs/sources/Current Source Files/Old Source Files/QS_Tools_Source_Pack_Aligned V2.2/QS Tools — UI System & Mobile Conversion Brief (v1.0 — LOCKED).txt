## 1. Audit Script (Phase 3)

### `mobile_ui_audit.py`

```python
import os
import re
import json

ROOT = r"D:\QSTools\qstools-web"

patterns = {
    "grid_cols_2plus": r"grid-cols-[2-9]",
    "fixed_width": r"max-w-|w-\[",
    "fixed_height": r"h-\[|min-h-\[",
    "small_touch_target": r"py-2|h-8|h-9",
    "tiny_text": r"text-xs",
}

results = []

for root, _, files in os.walk(ROOT):
    for file in files:
        if file.endswith((".jsx", ".js", ".tsx")):
            path = os.path.join(root, file)
            with open(path, "r", encoding="utf-8") as f:
                content = f.read()

            file_findings = []

            for name, pattern in patterns.items():
                matches = re.findall(pattern, content)
                if matches:
                    file_findings.append((name, len(matches)))

            if file_findings:
                results.append({
                    "file": path,
                    "issues": file_findings
                })

with open("audit_phase3.json", "w") as f:
    json.dump(results, f, indent=2)

print("Audit complete")
```

## 2. Layout Codemod (Phase 3 Starter)

### `ui_layout_codemod_v3.py`

```python
import os
import re

ROOT = r"D:\QSTools\qstools-web"

def fix_grids(content):
    content = re.sub(r"md:grid-cols-\d+", "", content)
    content = re.sub(r"xl:grid-cols-\d+", "", content)
    return content

def fix_text(content):
    content = content.replace("text-xs", "text-sm")
    return content

def process_file(path):
    with open(path, "r", encoding="utf-8") as f:
        original = f.read()

    updated = original
    updated = fix_grids(updated)
    updated = fix_text(updated)

    if updated != original:
        with open(path, "w", encoding="utf-8") as f:
            f.write(updated)
        return True

    return False

for root, _, files in os.walk(ROOT):
    for file in files:
        if file.endswith((".jsx", ".js", ".tsx")):
            path = os.path.join(root, file)
            changed = process_file(path)
            if changed:
                print(f"Updated: {path}")

print("Codemod complete")
```
